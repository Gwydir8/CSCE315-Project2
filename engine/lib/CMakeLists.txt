set(SRCS
  engine.cc
  relation.cc
  utility.cc
  parser.cc
	)

add_library(${ENGINE_LIBRARY} SHARED ${SRCS})
add_library(${ENGINE_LIBRARY_DEBUG} SHARED ${SRCS})

#use C++14
set_property(TARGET ${ENGINE_LIBRARY} ${ENGINE_LIBRARY_DEBUG} PROPERTY CXX_STANDARD 14)
#C++14 required
set_property(TARGET ${ENGINE_LIBRARY} ${ENGINE_LIBRARY_DEBUG} PROPERTY CXX_STANDARD_REQUIRED ON)

set_target_properties(${ENGINE_LIBRARY} PROPERTIES COMPILE_FLAGS "${CXXFLAGS_RELEASE}")
set_target_properties(${ENGINE_LIBRARY_DEBUG} PROPERTIES COMPILE_FLAGS "${CXXFLAGS_DEBUG}")
set_target_properties(${ENGINE_LIBRARY} PROPERTIES VERSION 1.0.0 SOVERSION 1)
set_target_properties(${ENGINE_LIBRARY_DEBUG} PROPERTIES VERSION 1.0.0 SOVERSION 1)
